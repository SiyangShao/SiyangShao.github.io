CF1542(Round 729)⌗

B. Plus and Multiply⌗

题意⌗

定义一个集合$S$，满足如下条件 $$ 1\in S $$ $$ x\in S \to x*a \in S , x+b \in S $$ 给出$n,a,b$，问$n$是否在集合中

算法⌗

显然的，对于满足该条件的$n$而言，有 $$ x = a^p + b*q $$ 于是可以枚举$a$判断
需要注意的是，对于$1$，需要特判存在

代码⌗ 

#include <bits/stdc++.h>
using namespace std;
#define int long long
int x, a, b;
inline bool judge() {
    if(x==1||b==1||x%b==1) return true;
    if(a==1){
        if(x%b==1) return true;
        else return false;
    }
    for (int base = 1 , i = 1; i<=30,base <= x; ++i,base *= a) {
        if ((x - base) % b == 0) {
            return true;
        }
    }
    return false;
}
signed main() {
    int _;
    scanf("%lld",&_);
    while (_--) {
        scanf("%lld %lld %lld", &x, &a, &b);
        if (judge()) {
            printf("Yes\n");
        }
        else {
            printf("No\n");
        }
    }
}

C.Strange Function⌗

题意⌗

定义函数$f(x)$，其中$f(x)$为最小的非$x$因数的数，求对于给定的$n$ $$ \sum_{i=1}^{n}f(i) $$ 显然的，给出的$f(n)$值并不大。同样显然的，若$n%i==0$，则$f(i)>=f(n)$。计算得到，$f(i)_{max} = 41$，则可以递推求出$a_n$，其中$a_n = \min(f(i)==a_n)$，即可得知$a_i=LCM(1,2,3,\cdots,i)$而后即可计算得出$sum$值。
其中，$a_n$可以通过递推求解

inline void pre(){
    a[0]=a[1]=1;
    for(int j = 1 ; j < cnt ; ++j){
        for(int i = 1 ; i<=j ; ++i){
            if(a[j-1]*i%j==0){
                a[j]=a[j-1]*i;
                break;
            }
        }
    }
}

于是可得， $$ ans=(n+\sum_{i=1}^{a_i\leq n}\lfloor\frac{n}{a_i}\rfloor) $$

代码⌗ 

#include <bits/stdc++.h>
#define int long long
using namespace std;
const int mod = 1e9+7 , maxn = 1e16+7,cnt=41;
int a[50];
inline void pre(){
    a[0]=a[1]=1;
    for(int j = 1 ; j < cnt ; ++j){
        for(int i = 1 ; i<=j ; ++i){
            if(a[j-1]*i%j==0){
                a[j]=a[j-1]*i;
                break;
            }
        }
    }
}
signed main(){
    int _,n;
    pre();
    scanf("%lld",&_);
    while(_--){
        scanf("%lld",&n);
        int ans = 0;
        for(int i = 0 ; i < cnt && a[i] <= n ; ++i){
            ans += n/a[i];
            ans %= mod;
        }
        printf("%lld\n",ans);
    }
}
