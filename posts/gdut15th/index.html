A.M形字符串⌗

题意：⌗

给出一个长度为n的字符串，求解其有多少个前缀是M形字符串（两个相同的回文串拼接）

算法：⌗

注意到若一个字符串是M形字符串，则字符串$[0,i]$和字符串$[0,\frac{i+1}{2}]$必为回文串

则可通过字符串Hash预处理并判断

对字符串正反Hash预处理（自然溢出）：

void makeHash() {
    s[0] = 1;
    for (int i = 1; i < N; i++) s[i] = p * s[i - 1];
    for (int i = 1; i < N; i++) hashL[i] = hashL[i - 1] * p + m[i] - 'a';
    for (int i = n; i >= 1; i--) hashR[n + 1 - i] = hashR[n - i] * p + m[i] - 'a';
}

判断该字符串是否为回文串：

bool check(int st, int ed) {
    ull t1 = hashL[ed] - hashL[st - 1] * s[ed - st + 1];
    ull t2 = hashR[n + 1 - st] - hashR[n - ed] * s[ed - st + 1];
    if (t1 == t2)return true;
    else return false;
}

代码⌗ 

#include<bits/stdc++.h>

using namespace std;
typedef unsigned long long ull;
const int N = 2e5 + 10;
const int p = 133;
char m[N];
int n;
ull hashL[N], hashR[N], s[N];

void makeHash() {
    s[0] = 1;
    for (int i = 1; i < N; i++) s[i] = p * s[i - 1];
    for (int i = 1; i < N; i++) hashL[i] = hashL[i - 1] * p + m[i] - 'a';
    for (int i = n; i >= 1; i--) hashR[n + 1 - i] = hashR[n - i] * p + m[i] - 'a';
}


bool check(int st, int ed) {
    ull t1 = hashL[ed] - hashL[st - 1] * s[ed - st + 1];
    ull t2 = hashR[n + 1 - st] - hashR[n - ed] * s[ed - st + 1];
    if (t1 == t2)return true;
    else return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    cin >> m + 1;
    n = strlen(m + 1);
    makeHash();
    int res = 0;
    for (int i = 1; i <= n; i++) {
        if (check(1, i)) {
            if (check(1, (i + 1) / 2)) {
                res++;
            }
        }
    }
    cout << res << "\n";
    return 0;
}

B.找山坡⌗

题意：⌗

找到使得$R-L$最大的$R$和$L$，满足$a[R]==a[L]$且对$i\in [L,R]$有$a[i]\geq a[L]$

算法：⌗

构建一个内部由小到大排序的堆栈使栈顶为min，对$a[i]$做如下处理：

若$a[i]<top$，则弹出$top$直到$a[i]\geq top$，并将$a[i]$入栈

若$a[i]>top$，将$a[i]$入栈

若$a[i]=top$，$ans=max(i-top)$

即：

	for (int i = 1; i <= n; i++) {
		while (top && a[st[top]] > a[i])
			top--;

		if (top && a[st[top]] == a[i])
			ans = max(ans, i - st[top]);
		else
			st[++top] = i;
	}

代码⌗ 

#include <algorithm>
#include <cstdio>
#include <iostream>
using namespace std;
const int maxn = 1e6 + 10;

int a[maxn];
int st[maxn];
int top;

int main() {
	int n;
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) {
		scanf("%d", &a[i]);
	}

	int ans = 0;

	for (int i = 1; i <= n; i++) {
		while (top && a[st[top]] > a[i])
			top--;

		if (top && a[st[top]] == a[i])
			ans = max(ans, i - st[top]);
		else
			st[++top] = i;
	}

	printf("%d\n", ans);

	return 0;
}

C.涂墙⌗

题意：⌗

给出一个自然数n，判断n是否能表示成五个正整数平方和

算法1：⌗

DFS减支（较low）

void dfs(int s, int p) {
    if (p != 5 && s <= 0) return;
    if (p > 5) return;
    if (p == 5 && s == 0) {
        cout << "YES" << endl;
        flag = 1;
        return;
    }
    int x = pow(s, 0.5);
    for (int i = x; i >= 1; i--) {

        flag = 0;
        s -= i * i;
        dfs(s, p + 1);
        if (flag) break;
        s += i * i;
    }


}

算法2：⌗

由拉格朗日平方和定理：

每个正整数都可以表示为四个平方数（包括0）的和

注意到169可以表示为1个或2个或3个或4个平方数的和 $$ 169=13^2\
169=12^2+5^2\
169=12^2+4^2+3^2\
169=10^2+8^2+2^2+1^2\
169=12^2+4^2+2^2+2^2+1^2 $$ 所以对一个不小于169的正整数，必然满足条件（n-169可以用1~4个表示，必然能凑出五个）

对于小于169的正整数，发现0,1,2,3,4,6,7,9,10,12,15,18,33不满足

代码⌗ 

#include <bits/stdc++.h>
using namespace std;
int main() {
	int t;
	cin >> t;
	while (t--) {
		int n;
		cin >> n;
		if (n == 0 || n == 1 || n == 2 || n == 3 || n == 4 || n == 6 ||
			n == 7 || n == 9 || n == 10 || n == 12 || n == 15 || n == 18 ||
			n == 33) {
			cout << "NO\n";
		} else {
			cout << "YES\n";
		}
	}
}

G.分割⌗

题意：⌗

给出n条平行于y轴的直线和m条平行于x轴的直线，求他们可能形成的矩形的和

算法：⌗

$$ Sum = \Sigma{(x_i-x_{i-1})\cdot(y_i-y_{i-1})}\
Sum = \Sigma{(x_i\cdot{(2i-1-n)})}\cdot\Sigma(y_i\cdot{(2i-1-m)}) $$

代码⌗ 

#include <bits/stdc++.h>
using namespace std;
int n,m;
long long sumx , sumy;
const long long mod = 1000000007;
long long x[100001] , y[100001];
bool cmp(long long a , long long b){
    return a<b;
}
int main(){
    cin>>n>>m;
    for(int i = 1 ; i <= n ; i++){
        cin>>x[i];
        x[i]+=1000000000;
    }
    sort(x+1,x+1+n,cmp);
    for(int i =1 ; i<= n ; i++){
        sumx += x[i]*(2*i-1-n) %mod;
        sumx%=mod;
    }
    for(int i = 1 ; i <= m ; i++){
        cin>>y[i];
        y[i]+=1000000000;
    }
    sort(y+1,y+1+m,cmp);
    for(int i = 1 ; i <= m ; i++){
        sumy += y[i]*(2*i-1-m) %mod;
        sumy%=mod;
    }
    cout<<sumx*sumy%mod<<"\n";
}

I.母牛哥与子序列⌗

题意:⌗

计算一个子序列所有非空子序列的积的和

算法:⌗

$$ Ans = Ans\cdot(a_i+1)+a_i $$

代码⌗ 

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll mod = 1000000007;
ll n , a , ans;
int main(){
    cin>>n>>a;
    ans = a;
    for(ll i = 2 ; i<= n ; i++){
        cin>>a;
        ans = ((ans)*(a+1))%mod+a;
        ans %= mod;
    }
    cout<<ans<<"\n";
}

E.捡贝壳⌗

题意：⌗

给出一个数列，多次查询，其中$[l,r]$里有多少个$x$的整数倍

P.S 一开始数据较弱，完全彻底的暴力搜索（遍历）可以过掉，复杂度差不多是1e9数量级？

算法：⌗

用vector存储大小为i的数字的位置

即：q[t]={a , b , c} 表现大小为t的数字的形式

那么可以用二分查找找到大小为t的数字在l到r区间内出现了多少次

tmp = upper_bound(a[i].begin(),a[i].end(),r)-lower_bound(a[i].begin(),a[i].end(),l);

遍历x的整数倍累加就可以得到解答

代码⌗ 

#include<bits/stdc++.h>

using namespace std;
typedef long long ll;
vector<int> a[100010];

int main() {
    int n, q, t,maxn = 0;
    cin >> n >> q;
    for (int i = 1; i <= n; i++) {
        scanf("%d", &t);
        a[t].push_back(i);
        maxn = max(maxn , t);
    }
    int l, r, x;
    while (q--) {
        int ans = 0;
        scanf("%d%d%d", &l, &r, &x);
        for (int i = x; i <= maxn; i += x) {
            int temp = upper_bound(a[i].begin(), a[i].end(), r) - lower_bound(a[i].begin(), a[i].end(), l);
            ans += temp;
        }
        cout << ans << endl;
    }
    return 0;
}
