A. Array and Peaks

题意

给出一个字符串，使其具有k个波峰

算法

略

代码

#include <bits/stdc++.h>
using namespace std;
int _, n, k;
int main() {
	cin >> _;
	while (_--) {
		cin >> n >> k;
		if (k > (n - 1) / 2) {
			cout << "-1\n";
		} else {
			vector<int> q;
			int i = 1;
			while (k--) {
				q.push_back(i);
				q.push_back(n + 1 - i);
				i++;
			}
			for (auto j : q) {
				cout << j << " ";
			}
			for (int j = i; j <= n + 1 - i; j++) {
				cout << j << " ";
			}
			cout << "\n";
		}
	}
}

B . AND Sequences

题意

给出一组字符串，问其有多少个排列能满足：

对$\forall x \in [1,n-1]$，有
$$
a_1 & a_2 &a_3\cdots&a_i=a_{i+1}&a_{i+2}&\cdots&a_n
$$

算法

考虑到位运算$&$的性质有$a&a==a$，令$a_1&a_2&\cdots&a_n=a$，有$a_i&a==a$，$a_1=a$，$a_n=a$

枚举判断即可

代码

#include <bits/stdc++.h>
using namespace std;
#define int long long
const int mod = 1e9 + 7;
int _, n, ans, an, tmp;
int a[200010];

signed main() {
	cin >> _;
	while (_--) {
		cin >> n;
		cin >> a[1];
		an = a[1];
		ans = 0;
		for (int i = 2; i <= n; i++) {
			cin >> a[i];
			an = (an & a[i]);
		}
		bool flag = true;
		for (int i = 1; i <= n; i++) {
			if (an == a[i]) {
				ans++;
			}
			if ((an & a[i]) != an) {
				flag = false;
				break;
			}
		}
        if(ans<2) flag = false;
		if (flag) {
			tmp = ans * (ans - 1);
			for (int i = 1; i <= n - 2; i++) {
				tmp *= i;
				tmp %= mod;
			}
			cout << tmp << "\n";
		}else{
            cout<<"0\n";
        }
	}
}

C . Add One

题意

给出一个数字$n$，对他进行$m$次操作，每次操作为对其所有数位+1

算法

dfs记忆化搜索存储9加m次后有几位暴力即可

代码

#include <bits/stdc++.h>
#define int long long
using namespace std;
const int mod = 1e9+7;
string s ;
int m , _ ;
int f[200010] ;
int judge(int x , int y){
    if(y==0) return 1;
    if(x+y<10){
        return 1;
    }
    if(x+y<=18){
        return 2;
    }
    if(x+y==19) return 3;
    if(x+y<=27) return 4;
    if(x+y==28) return 5;
    if(x+y==29) return 7;
    if(x+y==30) return 8;
    if(f[y-(10-x)]==0){
        f[y-(10-x)] = judge(1,y-10+x)+judge(0,y-10+x);
        f[y-(10-x)]%=mod;
    }
    return f[y-(10-x)];
}
signed main(){
    ios::sync_with_stdio(false);
    cin.tie(0),cout.tie(0);
    cin>>_;
    judge(9,200000);
    while(_--){
        cin>>s>>m;
        int dp = 0;
        for(int i = 0 ; i < s.length() ; i++){
            int qs = s[i]-'0';
            dp += judge(qs,m);
            dp%=mod;
        }
        cout<<dp<<"\n";
    }
}

D . GCD and MST

题意

给出一串数字，其中相邻数字之间cost为p，而一串数字若其最小公因数是其中最小的数字，则其最左端与最右端cost为最小的数字，求最小生成树

算法

从最小值开始双端枚举，将每一串数字合并为其中代价均为最小值的单个节点

代码

#include <bits/stdc++.h>
#define int long long
using namespace std;
struct node {
	int n, i;
} a[200010];
int _, p, n, b[200010], vis[200010];
bool cmp(node x, node y) { return x.n < y.n; }
signed main() {
	ios::sync_with_stdio(false);
	cin.tie(0), cout.tie(0);
	cin >> _;
	while (_--) {
		memset(vis, 0, sizeof(vis));
		cin >> n >> p;
		for (int i = 1; i <= n; i++) {
			cin >> b[i];
			a[i].n = b[i];
			a[i].i = i;
		}
		sort(a + 1, a + 1 + n, cmp);
		int ans = 0, rest = n - 1;
		for (int i = 1; i <= n; i++) {
			if (a[i].n > p)
				break;
			if (vis[a[i].i])
				continue;
			int l = a[i].i, r = a[i].i;
			while (l > 1 && !vis[l] && (b[l - 1] % a[i].n == 0))
				l--;
			while (r < n && !vis[r] && (b[r + 1] % a[i].n == 0))
				r++;
			for (int i = l; i <= r; i++) {
				vis[i] = 1;
			}
			rest -= r - l;
			ans += (r - l) * a[i].n;
		}
		ans += rest * p;
		cout << ans << "\n";
	}
}
