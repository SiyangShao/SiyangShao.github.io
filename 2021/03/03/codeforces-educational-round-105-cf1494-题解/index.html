C. 1D Sokoban

题目链接

题意：

从（0，0）将an向左or向右移动，最多能让多少个an与bm重合

算法

当a和b都大于0时：

可用一个数组表示从0~i，有多少个an与bm是初始重合的

（p表示a，sp表示b，下同）

map<long,bool>mp;
for(long k:p) mp[k]=true;
long n = p.size() , m = sp.size();
vector<long> cum(m+1,0);
for(int i = 1; i<= m ; i++){
    cum[i]=cum[i-1]+mp[sp[i-1]];
}

然后接下来利用upper_bound函数查找出推箱子到b[i]点的最多连续个数

for(int i = 0 ; i < m ;i++){
    long can = cum[m] - cum[i+1];//can初始化为后续点中本来就为特殊点的个数
    auto k = upper_boound(p.begin(),p.end(),sp[i]);
    if(k==p.end()){
        long tot1 = n ;
    }else{
        long tot1 = k - p.begin();
        //tot1表示普通点中比第i个特殊点小的点的个数，即可推到的最长长度
    }
    auto l = upper_bound(sp.begin(),sp.end(),sp[i]-tot1);
    tot2 = l - sp.begin();//tot2表示不在从b[i]往前推的这一长串中的点的数目
    can += i + 1 - tot2;//can加上总共i+1个点，其中前tot2个不被覆盖
    mx = max(mx,can);
}

总代码：

#include <bits/stdc++.h>

using namespace std;

long solve(vector<long> &p, vector<long> &sp) {
    sort(p.begin(), p.end());
    sort(sp.begin(), sp.end());
    map<long, bool> mp;
    for (long k:p) mp[k] = true;
    long n = p.size(), m = sp.size(), i, mx = 0, can, tot1, tot2;
    vector<long> cum(m + 1, 0);
    for (i = 1; i <= m; i++) {
        cum[i] = cum[i - 1] + mp[sp[i - 1]];
    }
    for (i = 0; i < m; i++) {
        can = cum[m] - cum[i + 1];//通过can算出后续列表中位于特殊点的
        //cout<<i<<"  "<<can<<endl;
        auto k = upper_bound(p.begin(), p.end(), sp[i]);//比第i个特殊点更大
        if (k == p.end()) { tot1 = n; }
        else { tot1 = (k - p.begin()); }//tot:普通点中比第i个特殊点大的最小的点的数目
        auto l = upper_bound(sp.begin(), sp.end(), sp[i] - tot1);//sp[i]-tot1为最长的理论值
        tot2 = (l - sp.begin());//不在这一序列中的点的数目
        can += i + 1 - tot2;//结果
        //cout<<i<<"  "<<can<<endl;
        mx = max(mx, can);
    }
    return mx;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0),cout.tie(0);
    long t;
    cin >> t;
    while (t--) {
        long n, m, x;
        cin >> n >> m;
        vector<long> npos, pos, nspos, spos;
        while (n--) {
            cin >> x;
            if (x > 0) { pos.push_back(x); }
            else { npos.push_back(-x); }
        }
        while (m--) {
            cin >> x;
            if (x > 0) { spos.push_back(x); }
            else { nspos.push_back(-x); }
        }
        cout << solve(pos, spos) + solve(npos, nspos) << endl;
    }
}

D. Dogeforces

题目链接

题意：

给出所有子节点两个点之间的共有祖先的值，构造出原数

算法：

并查集+分治，对所有以值为i的点为根的点进行遍历，构造出子数

for (int i = 1; i <= mx - 1; i++) { //根节点的值一定大于总节点数目，遍历有同一根节点的点
     if (h[i].empty()) continue;//这个节点莫得
     int cur = k;//当前的值
     for (auto &j : h[i]) fa[j.first] = find(j.first), fa[j.second] = find(j.second); //初始化每一个的父节点
     for (auto &j : h[i]) f[find(fa[j.first])] = find(fa[j.second]);//合并
     for (auto &j : h[i]) {
         int x = find(j.first);
         if (x <= cur) b[++k] = i, f[x] = f[k] = k, x = k;//构造树
         a[fa[j.first]] = a[fa[j.second]] = x;
     }

代码：

#include<bits/stdc++.h>

using namespace std;
const int maxn = 50000;
int b[maxn], f[maxn], a[maxn], fa[maxn];
int k, n;
vector<pair<int, int> > h[maxn];

int find(int x) {
    return x == f[x] ? x : f[x] = find(f[x]); //并查集
}

int main() {
    cin >> n;
    k = n; //至少k个点
    int mx = 0; //根节点的值
    for (int i = 1; i <= n; i++) f[i] = i; //并查集初始化
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            int m;
            cin >> m;
            if (i == j) { b[i] = m; }
            else if (i < j) {
                h[m].emplace_back(i, j);
                mx = max(mx, m);
            }
        }
    }
    for (int i = 1; i <= mx - 1; i++) { 
        if (h[i].empty()) continue;
        int cur = k;
        for (auto &j : h[i]) fa[j.first] = find(j.first), fa[j.second] = find(j.second); 
        for (auto &j : h[i]) f[find(fa[j.first])] = find(fa[j.second]);
        for (auto &j : h[i]) {
            int x = find(j.first);
            if (x <= cur) b[++k] = i, f[x] = f[k] = k, x = k;
            a[fa[j.first]] = a[fa[j.second]] = x;
        }
    }
    b[++k] = mx;
    for (auto &i : h[mx]) {
        a[find(i.first)] = a[find(i.second)] = k;
    }
    cout << k << endl;
    for (int i = 1; i <= k; i++) cout << b[i] << " ";
    cout << endl;
    cout << k << endl;
    for (int i = 1; i <= k - 1; i++) cout << i << " " << a[i] << endl;

    return 0;
}

E. A-Z Graph

题目链接

题意

n个点，m次查询+修改。可令u->v=c，也可删去。查询是否能走出一个长度为k的回文路径（即从头到尾和从尾到头一样）

题解

本题目极水……早知道做题的时候就来写这道题目了….

注意到k为奇数时走偶数次，即u->v->u，必然是回文路径

k为偶数时需要u->v=v->u即可

if(c=='+'){
          int u , v;
          cin>>u>>v>>w;
          e[mp(u,v)]=w;
          pii g = mp(v,u);
          p+=(e.find(g)!=e.end());
          q+=(e.find(g)!=e.end()&&e[g]==w);

      }else if(c=='-'){
          int u , v;
          cin>>u>>v;
          w=e[mp(u,v)];
          e.erase(mp(u,v));
          pii g = mp(v,u);
          p-=(e.find(g)!=e.end());
          q-=(e.find(g)!=e.end()&&e[g]==w);
      }
//p和q分别表示u->v和v->u都存在的路径条数&&u->v==v->u的路径条数

代码

#include <bits/stdc++.h>
#define pii pair<int,int>
#define mp make_pair
using namespace std;
int n , m , q, p;
map<pair<int,int>,char>e;
int main(){
    cin>>n>>m;
    for(int i = 0 ; i < m ; i++){
        char c,w;
        cin>>c;
        if(c=='+'){
            int u , v;
            cin>>u>>v>>w;
            e[mp(u,v)]=w;
            pii g = mp(v,u);
            p+=(e.find(g)!=e.end());
            q+=(e.find(g)!=e.end()&&e[g]==w);

        }else if(c=='-'){
            int u , v;
            cin>>u>>v;
            w=e[mp(u,v)];
            e.erase(mp(u,v));
            pii g = mp(v,u);
            p-=(e.find(g)!=e.end());
            q-=(e.find(g)!=e.end()&&e[g]==w);
        }else{
            int k;
            cin>>k;
            if(k%2==1){
                if(p>0){
                    cout<<"YES\n";
                }else{
                    cout<<"NO\n";
                }
            }else{
                if(q>0){
                    cout<<"YES\n";
                }else{
                    cout<<"NO\n";
                }
            }
        }
    }
}
