源代码（python3.8）

'''
数据填写于data.yaml中，填写格式参照data.yaml中注释
由于yaml原因（因为作者懒且菜）
仅输出需要输入的数据，顺序均为按照填写顺序（即由1到n），其中d均为delta之意，后有%均代表为百分数
均输出于answer.yaml中
data.yaml中原始数据为作者舍友手测，answer.yaml中数据为作者由本程序算的结果，经检验与作者舍友手算数据无明显偏差
python版本为3.8
使用方式：填写数据完后python3(或者python?) main.py
或者直接采用pycharm打开本文件并点击运行
标准编码格式为UTF-8
不保证在其他环境下仍可运行
作者:shaosy
邮箱:siyang.shao@outlook.com
日期:2021/1/14
'''
import yaml
import math


# 由于python3 round函数存在精度丢失问题，故实现四舍五入,默认均保留四位小数：
def round_up(number, power=0):
    """
    实现精确四舍五入，包含正、负小数多种场景
    :param number: 需要四舍五入的小数
    :param power: 四舍五入位数，支持0-∞
    :return: 返回四舍五入后的结果
    """
    digit = 10 ** power
    num2 = float(int(number * digit))
    # 处理正数，power不为0的情况
    if number >= 0 and power != 0:
        tag = number * digit - num2 + 1 / (digit * 10)
        if tag >= 0.5:
            return (num2 + 1) / digit
        else:
            return num2 / digit
    # 处理正数，power为0取整的情况
    elif number >= 0 and power == 0:
        tag = number * digit - int(number)
        if tag >= 0.5:
            return (num2 + 1) / digit
        else:
            return num2 / digit
    # 处理负数，power为0取整的情况
    elif power == 0 and number < 0:
        tag = number * digit - int(number)
        if tag <= -0.5:
            return (num2 - 1) / digit
        else:
            return num2 / digit
    # 处理负数，power不为0的情况
    else:
        tag = number * digit - num2 - 1 / (digit * 10)
        if tag <= -0.5:
            return (num2 - 1) / digit
        else:
            return num2 / digit


with open('data.yaml') as file:
    data = yaml.safe_load(file)
answer = open('answer.yaml', 'w')
# 表格1数据处理，根据老师要求增加两次相差百分比计算
t71 = [data['t11'], data['t12'], data['t13']]
d1 = [0, 0, 0]
d2 = [0, 0, 0]
for i in range(3):
    d1[i] = round_up((t71[i][1] - t71[i][0]) * 100 / t71[i][0], 2)
    d2[i] = round_up((t71[i][3] - t71[i][2]) * 100 / t71[i][3], 2)
yaml.safe_dump_all(["table1 answer", "t1-t2(%)", d1, "dt1-dt2(%)", d2], answer)
# 表格2数据处理
t72 = [data['t21'], data['t22'], data['t23'], data['t24'], data['t25'], data['t26']]
dt = [0, 0, 0, 0]
st = [0, 0, 0, 0]
ut = [0, 0, 0, 0]
for i in range(4):
    for j in range(6):
        dt[i] = dt[i] + t72[j][i]
    dt[i] = round_up(dt[i] / 6, 4)
    for j in range(6):
        st[i] = st[i] + (t72[j][i] - dt[i]) * (t72[j][i] - dt[i])
    st[i] = st[i] / 5
    st[i] = round_up(math.sqrt(st[i]), 4)
    ut[i] = round_up(math.sqrt(st[i] * st[i] + 0.05 * 0.05), 4)
yaml.safe_dump_all(["table2 answer", "dt", dt, "st", st, "ut", ut], answer)
# 表格3数据处理
t31 = [1, 86, 60, 1]
t32 = [0.002, 0.05, 0.05, 0.002]
E = [0, 0, 0, 0]
for i in range(4):
    E[i] = round_up(t32[i] * 100 / t31[i], 2)
yaml.safe_dump_all(["table 3", "E(%)", E], answer)
# 第三小问
Tb = round_up((1 / (dt[1] * dt[1]) + 1 / (dt[2] * dt[2])) - (1 / (dt[0] * dt[0]) + 1 / (dt[3] * dt[3])), 4)
g = (t31[1] * t31[0] * t31[0] / (4 * t31[2] * t31[3])) * Tb * 10000
Eg = 4 * E[0] * E[0] / 10000 + E[1] * E[1] / 10000 + E[2] * E[2] / 10000 + E[3] * E[3] / 10000
for i in range(4):
    Eg += 4 * (ut[i] * ut[i] / (math.pow(dt[i], 6))) / (Tb * Tb)
Eg = round_up(math.sqrt(Eg), 4)
Ug = round_up(g * Eg, 4)
yaml.safe_dump_all(["Else", "T", Tb, "g", g, "Eg", Eg, "Ug", Ug], answer)
# 实验结果
Ag = round_up(abs(g - 9.794) * 100 / 9.794, 2)
Eg = round_up(Eg * 100, 2)
yaml.safe_dump_all(["Ag", Ag], answer)
